341void MPU6050_Init(void)
342{
343 uint8_t check;
344 uint8_t data;
345 char buffer[100];
346
347 // Check WHO_AM_I register (should return 0x68)
348 HAL_I2C_Mem_Read(&hi2c2, MPU6050_ADDR, WHO_AM_I_REG, 1, &check, 1, HAL_MAX_DELAY);
349
350 if (check == 0x68) // MPU6050 WHO_AM_I value
351 {
352 sprintf(buffer, "MPU6050 detected! WHO_AM_I = 0x%02X\r\n", check);
353 UART_Print(buffer);
354
355 // Wake up the MPU6050 (clear sleep bit in power management register)
356 data = 0x00;
357 HAL_I2C_Mem_Write(&hi2c2, MPU6050_ADDR, PWR_MGMT_1_REG, 1, &data, 1, HAL_MAX_DELAY);
358
359 // Set sample rate to 1kHz
360 data = 0x07;
361 HAL_I2C_Mem_Write(&hi2c2, MPU6050_ADDR, SMPLRT_DIV_REG, 1, &data, 1, HAL_MAX_DELAY);
362
363 // Configure accelerometer (+/- 2g)
364 data = 0x00;
365 HAL_I2C_Mem_Write(&hi2c2, MPU6050_ADDR, ACCEL_CONFIG_REG, 1, &data, 1, HAL_MAX_DELAY);
366
367 // Configure gyroscope (+/- 250 deg/s)
368 data = 0x00;
369 HAL_I2C_Mem_Write(&hi2c2, MPU6050_ADDR, GYRO_CONFIG_REG, 1, &data, 1, HAL_MAX_DELAY);
370
371 UART_Print("MPU6050 initialized successfully!\r\n");
372 }
373 else
374 {
375 sprintf(buffer, "MPU6050 not found! WHO_AM_I = 0x%02X\r\n", check);
376 UART_Print(buffer);
377 }
378}

384void MPU6050_Read_All(void)
385{
386 uint8_t rec_data[14];
387 int16_t accel_x, accel_y, accel_z;
388 int16_t gyro_x, gyro_y, gyro_z;
389 int16_t temperature;
390 char buffer[200];
391
392 // Read 14 bytes starting from accelerometer high byte register
393 HAL_I2C_Mem_Read(&hi2c2, MPU6050_ADDR, ACCEL_XOUT_H_REG, 1, rec_data, 14, HAL_MAX_DELAY);
394
395 // Combine high and low bytes for each reading
396 accel_x = (int16_t)(rec_data[0] << 8 | rec_data[1]);
397 accel_y = (int16_t)(rec_data[2] << 8 | rec_data[3]);
398 accel_z = (int16_t)(rec_data[4] << 8 | rec_data[5]);
399
400 temperature = (int16_t)(rec_data[6] << 8 | rec_data[7]);
401
402 gyro_x = (int16_t)(rec_data[8] << 8 | rec_data[9]);
403 gyro_y = (int16_t)(rec_data[10] << 8 | rec_data[11]);
404 gyro_z = (int16_t)(rec_data[12] << 8 | rec_data[13]);
405
406 // Convert raw values to meaningful units
407 // Accelerometer: 2g range, 16384 LSB/g
408 float accel_x_g = accel_x / 16384.0;
409 float accel_y_g = accel_y / 16384.0;
410 float accel_z_g = accel_z / 16384.0;
411
412 // Gyroscope: 250/s range, 131 LSB/(/s)
413 float gyro_x_dps = gyro_x / 131.0;
414 float gyro_y_dps = gyro_y / 131.0;
415 float gyro_z_dps = gyro_z / 131.0;
416
417 // Temperature: (TEMP_OUT/340 + 36.53) C
418 float temp_c = (temperature / 340.0) + 36.53;
419
420 // Print raw data
421 sprintf(buffer, "Raw Data:\r\nAccel: X=%6d Y=%6d Z=%6d\r\n", accel_x, accel_y, accel_z);
422 UART_Print(buffer);
423 sprintf(buffer, "Gyro: X=%6d Y=%6d Z=%6d\r\n", gyro_x, gyro_y, gyro_z);
424 UART_Print(buffer);
425 sprintf(buffer, "Temp: %6d\r\n", temperature);
426 UART_Print(buffer);
427
428 // Print converted data
429 sprintf(buffer, "\r\nConverted Data:\r\nAccel: X=%6.2fg Y=%6.2fg Z=%6.2fg\r\n",
430 accel_x_g, accel_y_g, accel_z_g);
431 UART_Print(buffer);
432 sprintf(buffer, "Gyro: X=%7.1f Y=%7.1f Z=%7.1f (deg/s)\r\n",
433 gyro_x_dps, gyro_y_dps, gyro_z_dps);
434 UART_Print(buffer);
435 sprintf(buffer, "Temp: %.1fC\r\n", temp_c);
436 UART_Print(buffer);
437}
