265void I2C_Scanner(void)
266{
267 char buffer[100];
268 uint8_t devices_found = 0;
269 HAL_StatusTypeDef result;
270
271 UART_Print("Starting I2C scan...\r\n");
272 UART_Print("Address Status\r\n");
273 UART_Print("------- ------\r\n");
274
275 for(uint8_t address = 1; address < 128; address++)
276 {
277 // HAL_I2C_IsDeviceReady checks if device responds to its address
278 result = HAL_I2C_IsDeviceReady(&hi2c2, (address << 1), 1, 10);
279
280 if(result == HAL_OK)
281 {
282 sprintf(buffer, "0x%02X Found!\r\n", address);
283 UART_Print(buffer);
284 devices_found++;
285 }
286 else if(result == HAL_ERROR)
287 {
288 // Uncomment next line if you want to see all tested addresses
289 // sprintf(buffer, "0x%02X No response\r\n", address);
290 // UART_Print(buffer);
291 }
292 }
293
294 if(devices_found == 0)
295 {
296 UART_Print("\r\nNo I2C devices found!\r\n");
297 UART_Print("Check connections:\r\n");
298 UART_Print("- SCL connected to PB10\r\n");
299 UART_Print("- SDA connected to PB3\r\n");
300 UART_Print("- Pull-up resistors on SDA/SCL (4.7kΩ)\r\n");
301 UART_Print("- Device power supply\r\n");
302 }
303 else
304 {
305 sprintf(buffer, "\r\nScan complete! Found %d device(s).\r\n", devices_found);
306 UART_Print(buffer);
307 }
308}
